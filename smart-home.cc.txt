NS_LOG_COMPONENT_DEFINE("SmartHome_Jamming_RSSI");
static std::ofstream g_csv;

/* ================= GLOBALS ================= */
static uint64_t g_rxProbes = 0;

/* ===== ADDITION (ONLY) ===== */
static uint64_t g_txProbes = 0;
/* =========================== */

/* RSSI globals */
static double   g_rssiSum = 0.0;
static double   g_rssiSum = 0.0;
static uint32_t g_rssiCount = 0;
static std::vector<double> g_rssiBaseline;
/*pdr+delay+throughput globals*/

static std::vector<double> g_pdrBaseline;
static std::vector<double> g_thptBaseline;
static std::vector<double> g_delayBaseline;

static double g_pdrMean = 0.0, g_pdrStd = 0.0;
static double g_thptMean = 0.0, g_thptStd = 0.0;
static double g_delayMean = 0.0, g_delayStd = 0.0;

/* Probe baseline */
static std::vector<uint64_t> g_probeBaseline;

static const uint32_t BASELINE_SAMPLES = 25;
static const double   BASELINE_START   = 10.0;
static const uint32_t TIMEOUT_SEC      = 5;

static bool g_baselineDone = false;
static bool g_detectorReady = false;

/* Stats */
static double g_probeMean = 0.0, g_probeStd = 0.0;
static double g_rssiMean  = 0.0, g_rssiStd  = 0.0;

static uint32_t anomalyCounter = 0;
/* FlowMonitor */
static uint64_t g_lastTxPkts = 0;
static uint64_t g_lastRxPkts = 0;
static uint64_t g_lastRxBytes = 0;
static double   g_lastDelaySum = 0.0;

Ptr<FlowMonitor> g_flowmon;
FlowMonitorHelper g_flowmonHelper;

/* ================= CALLBACKS ================= */
void
ProbeRxCallback(Ptr<const Packet>, const Address &)
{
  g_rxProbes++;
}

/* ===== ADDITION (ONLY) ===== */
void
ProbeTxCallback(Ptr<const Packet>)
{
  g_txProbes++;
}
/* =========================== */

/* RSSI from PHY */
void
PhySnifferRx(Ptr<const Packet>,
             uint16_t,
             WifiTxVector,
             MpduInfo,
             SignalNoiseDbm signalNoise,
             uint16_t)
{
  g_rssiSum += signalNoise.signal;
  g_rssiCount++;
}

/* ================= MONITOR ================= */
void
Monitor()
{
  static uint64_t lastProbeRx = 0;
  static uint64_t lastProbeTx = 0;

  uint64_t currProbeRx = g_rxProbes;
  uint64_t probeRxSec = currProbeRx - lastProbeRx;

  /* ===== ADDITION (ONLY) ===== */
  uint64_t currProbeTx = g_txProbes;
  uint64_t probeTxSec = currProbeTx - lastProbeTx;
  /* =========================== */

  double avgRssi = (g_rssiCount > 0)
                   ? g_rssiSum / g_rssiCount
                   : 0.0;
  std::cout << Simulator::Now().GetSeconds()
            << "s | ProbesTx=" << probeTxSec
            << " | ProbesRx=" << probeRxSec
            << " | RSSI=" << avgRssi << " dBm";

  /* ===== Baseline ===== */
  if (!g_baselineDone &&
      Simulator::Now().GetSeconds() >= BASELINE_START)
  {
    g_probeBaseline.push_back(probeRxSec);
    g_rssiBaseline.push_back(avgRssi);

    if (g_probeBaseline.size() == BASELINE_SAMPLES)
    {
      g_probeMean = std::accumulate(
        g_probeBaseline.begin(),
        g_probeBaseline.end(), 0.0) / g_probeBaseline.size();

      double varP = 0.0;
      for (auto v : g_probeBaseline)
        varP += std::pow(v - g_probeMean, 2);
      g_probeStd = std::sqrt(varP / g_probeBaseline.size());

      g_rssiMean = std::accumulate(
        g_rssiBaseline.begin(),
        g_rssiBaseline.end(), 0.0) / g_rssiBaseline.size();

      double varR = 0.0;
      for (auto v : g_rssiBaseline)
        varR += std::pow(v - g_rssiMean, 2);
      g_rssiStd = std::sqrt(varR / g_rssiBaseline.size());

/* ===== Delay stats ===== */
g_delayMean = std::accumulate(
  g_delayBaseline.begin(),
  g_delayBaseline.end(), 0.0) / g_delayBaseline.size();

double varD = 0.0;
for (auto v : g_delayBaseline)
  varD += std::pow(v - g_delayMean, 2);
g_delayStd = std::sqrt(varD / g_delayBaseline.size());

/* ===== Throughput stats ===== */
g_thptMean = std::accumulate(
  g_thptBaseline.begin(),
  g_thptBaseline.end(), 0.0) / g_thptBaseline.size();

double varT = 0.0;
for (auto v : g_thptBaseline)
  varT += std::pow(v - g_thptMean, 2);
g_thptStd = std::sqrt(varT / g_thptBaseline.size());

/* ===== PDR stats ===== */
g_pdrMean = std::accumulate(
  g_pdrBaseline.begin(),
  g_pdrBaseline.end(), 0.0) / g_pdrBaseline.size();
double varPdr = 0.0;
for (auto v : g_pdrBaseline)
  varPdr += std::pow(v - g_pdrMean, 2);
g_pdrStd = std::sqrt(varPdr / g_pdrBaseline.size());

      std::cout << "\n\n===== BASELINE =====\n";
      std::cout << "Probe μ=" << g_probeMean
                << " σ=" << g_probeStd << "\n";
      std::cout << "RSSI  μ=" << g_rssiMean
                << " σ=" << g_rssiStd << "\n";


std::cout << "Delay μ=" << g_delayMean
          << " σ=" << g_delayStd << "\n";
std::cout << "Thpt  μ=" << g_thptMean
          << " σ=" << g_thptStd << "\n";
std::cout << "PDR   μ=" << g_pdrMean
          << " σ=" << g_pdrStd << "\n";
std::cout << "====================\n\n";

      g_baselineDone = true;
      g_detectorReady = true;
    }
  }

  /* ===== FlowMonitor ===== */
  g_flowmon->CheckForLostPackets();
  auto stats = g_flowmon->GetFlowStats();
  uint64_t txPkts = 0, rxPkts = 0, rxBytes = 0;
  double delaySum = 0.0;

  for (const auto &f : stats)
  {
    txPkts   += f.second.txPackets;
    rxPkts   += f.second.rxPackets;
    rxBytes  += f.second.rxBytes;
    delaySum += f.second.delaySum.GetSeconds();
  }

  uint64_t flowTx = txPkts - g_lastTxPkts;
  uint64_t flowRx = rxPkts - g_lastRxPkts;

  double pdr = (flowTx > 0) ? (double)flowRx / flowTx : 0.0;
  double throughput = ((rxBytes - g_lastRxBytes) * 8.0) / 1e6;
  double delay = (rxPkts > g_lastRxPkts)
                 ? (delaySum - g_lastDelaySum) /
                   (rxPkts - g_lastRxPkts)
                 : 0.0;

  std::cout << " | PDR=" << pdr * 100 << "%"
            << " | Thpt=" << throughput << " Mbps"
            << " | Delay=" << delay * 1000 << " ms";

  g_lastTxPkts   = txPkts;
  g_lastRxPkts   = rxPkts;
  g_lastRxBytes  = rxBytes;
  g_lastDelaySum = delaySum;

if (!g_baselineDone &&
    Simulator::Now().GetSeconds() >= BASELINE_START)
{
  g_delayBaseline.push_back(delay);
  g_thptBaseline.push_back(throughput);
  g_pdrBaseline.push_back(pdr);
}


  /* ===== Detector ===== */
  if (g_detectorReady)
  {
    bool probeAnomaly =
      (probeRxSec < g_probeMean - 3*g_probeStd);

    bool rssiAnomaly =
      (avgRssi > g_rssiMean + 3*g_rssiStd);

     bool pdrAnomaly =
       (pdr < g_pdrMean - 3*g_pdrStd);
     bool delayAnomaly =
       (delay > g_delayMean + 3*g_delayStd);
     bool thptAnomaly =
       (throughput < g_thptMean - 3*g_thptStd);
      if (probeAnomaly || rssiAnomaly || pdrAnomaly || delayAnomaly || thptAnomaly)
{
  anomalyCounter++;
  std::cout << " ⚠(" << anomalyCounter << "/" << TIMEOUT_SEC << ")";
}
else
{
  anomalyCounter = 0;
}

    if (anomalyCounter >= TIMEOUT_SEC)
    {
      std::cout << "\n🚨 JAMMING DETECTED 🚨\n";
      anomalyCounter = 0;
    }
  }

  std::cout << std::endl;

  lastProbeRx = currProbeRx;
  lastProbeTx = currProbeTx;

  g_rssiSum = 0.0;
  g_rssiCount = 0;

  
g_csv << Simulator::Now().GetSeconds() << ","
      << probeTxSec << ","
     << probeRxSec << ","
      << avgRssi << ","
      << pdr << ","
      << throughput << ","
      << delay << std::endl;

Simulator::Schedule(Seconds(1.0), &Monitor);
}

/* ================= MAIN ================= */
int
main(int argc, char *argv[])
{
  double simTime = 301.0;

  Ptr<Node> router = CreateObject<Node>();
  NodeContainer stas;
  stas.Create(4);

  Ptr<Node> camera    = stas.Get(0);
  Ptr<Node> sensor    = stas.Get(1);
  Ptr<Node> appliance = stas.Get(2);
  Ptr<Node> attacker  = stas.Get(3);

  YansWifiChannelHelper chan = YansWifiChannelHelper::Default();
  YansWifiPhyHelper phy;
  phy.SetChannel(chan.Create());
  WifiHelper wifi;
  wifi.SetStandard(WIFI_STANDARD_80211g);
  wifi.SetRemoteStationManager("ns3::AarfWifiManager");

  WifiMacHelper mac;
  Ssid ssid("smart-home");

  mac.SetType("ns3::StaWifiMac",
              "Ssid", SsidValue(ssid),
              "ActiveProbing", BooleanValue(false));
  NetDeviceContainer staDevs = wifi.Install(phy, mac, stas);

  mac.SetType("ns3::ApWifiMac", "Ssid", SsidValue(ssid));
  NetDeviceContainer apDev = wifi.Install(phy, mac, router);

  /* Attach RSSI sniffer */
  for (uint32_t i = 0; i < staDevs.GetN(); ++i)
  {
    Ptr<WifiNetDevice> dev =
      DynamicCast<WifiNetDevice>(staDevs.Get(i));
    dev->GetPhy()->TraceConnectWithoutContext(
      "MonitorSnifferRx",
      MakeCallback(&PhySnifferRx));
  }

  /* Jammer PHY */
  Ptr<WifiNetDevice> jamDev =
    DynamicCast<WifiNetDevice>(staDevs.Get(3));
  jamDev->GetPhy()->SetTxPowerStart(30.0);
  jamDev->GetPhy()->SetTxPowerEnd(30.0);

  MobilityHelper mob;
  mob.SetMobilityModel("ns3::ConstantPositionMobilityModel");
  mob.Install(router);
  mob.Install(stas);

  InternetStackHelper stack;
  stack.Install(router);
  stack.Install(stas);

  Ipv4AddressHelper addr;
  addr.SetBase("10.1.1.0", "255.255.255.0");
  Ipv4InterfaceContainer apIf = addr.Assign(apDev);
  Ipv4InterfaceContainer staIf = addr.Assign(staDevs);

  Ipv4Address ipRouter = apIf.GetAddress(0);

  /* Probe sink */
  PacketSinkHelper sink(
    "ns3::UdpSocketFactory",
    InetSocketAddress(Ipv4Address::GetAny(), 9999));
  ApplicationContainer sinkApp = sink.Install(router);
  sinkApp.Start(Seconds(1.0));

  Ptr<PacketSink> ps =
    DynamicCast<PacketSink>(sinkApp.Get(0));
 ps->TraceConnectWithoutContext(
    "Rx", MakeCallback(&ProbeRxCallback));

  /* Probe Tx */
  OnOffHelper probeTx(
    "ns3::UdpSocketFactory",
    InetSocketAddress(ipRouter, 9999));
  probeTx.SetAttribute("DataRate", DataRateValue(DataRate("100kbps")));
  probeTx.SetAttribute("PacketSize", UintegerValue(64));
  probeTx.SetAttribute("OnTime",
    StringValue("ns3::ConstantRandomVariable[Constant=1]"));
  probeTx.SetAttribute("OffTime",
    StringValue("ns3::ConstantRandomVariable[Constant=0]"));

  ApplicationContainer probeApps;
  probeApps.Add(probeTx.Install(camera));
  probeApps.Add(probeTx.Install(sensor));
  probeApps.Add(probeTx.Install(appliance));

  for (uint32_t i = 0; i < probeApps.GetN(); ++i)
  {
    probeApps.Get(i)->TraceConnectWithoutContext(
      "Tx", MakeCallback(&ProbeTxCallback));
  }

  probeApps.Start(Seconds(2.0));

  /* Jammer traffic */
  OnOffHelper jammer(
    "ns3::UdpSocketFactory",
    InetSocketAddress(ipRouter, 4444));
  jammer.SetAttribute("DataRate", DataRateValue(DataRate("50Mbps")));
  jammer.SetAttribute("PacketSize", UintegerValue(32));
  jammer.SetAttribute("OnTime",
    StringValue("ns3::ConstantRandomVariable[Constant=1]"));
  jammer.SetAttribute("OffTime",
    StringValue("ns3::ConstantRandomVariable[Constant=0]"));

  jammer.Install(attacker).Start(Seconds(60.0));

g_csv.open("results.csv");
g_csv << "time,ProbesTx,ProbesRx,RSSI,PDR,Throughput,Delay\n";

  /* FlowMonitor */
  g_flowmon = g_flowmonHelper.InstallAll();

  Simulator::Schedule(Seconds(5.0), &Monitor);
  Simulator::Stop(Seconds(simTime));
  Simulator::Run();
  Simulator::Destroy();
g_csv.close();

  return 0;
}
